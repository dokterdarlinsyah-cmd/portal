// --- HELPER: SORT CATEGORIES (Newest/Largest ID first) ---
const sortCategoriesDesc = (cats) => {
  if (!Array.isArray(cats)) return [];
  return [...cats].sort((a, b) => {
    const timeA = String(a.id).startsWith('cat_') ? parseInt(String(a.id).split('_')[1] || 0, 10) : 0;
    const timeB = String(b.id).startsWith('cat_') ? parseInt(String(b.id).split('_')[1] || 0, 10) : 0;
    return timeB - timeA;
  });
};

// --- HELPER: SMART GOOGLE DRIVE LINK FIXER ---
const getEmbedUrl = (url) => {
  let targetUrl = url;
  try {
    if (targetUrl.includes('drive.google.com') || targetUrl.includes('docs.google.com') || targetUrl.includes('sheets.google.com') || targetUrl.includes('slides.google.com')) {
       // CASE A: Google Drive FOLDERS
       if (targetUrl.includes('/folders/')) {
          const folderIdMatch = targetUrl.match(/\/folders\/([a-zA-Z0-9_-]+)/);
          if (folderIdMatch && folderIdMatch[1]) {
             // Force list view for better embed experience
             return `https://drive.google.com/embeddedfolderview?id=${folderIdMatch[1]}#list`;
          }
       }
       // CASE B: Google Drive FILES (Documents, Spreadsheets, etc.)
       else {
           if (targetUrl.includes('/view')) {
               return targetUrl.replace(/\/view[^/?#]*/, '/preview');
           } else if (targetUrl.includes('/edit')) {
               return targetUrl.replace(/\/edit[^/?#]*/, '/preview');
           } else if (targetUrl.includes('/open')) {
               const idMatch = targetUrl.match(/id=([a-zA-Z0-9_-]+)/);
               if (idMatch && idMatch[1]) {
                   return `https://drive.google.com/file/d/${idMatch[1]}/preview`;
               }
           }
       }
    }
  } catch (e) {
    console.warn("Gagal memproses URL Google", e);
  }
  return targetUrl;
};

// --- STORAGE HELPERS ---
const getStorage = (key, fallback) => {
  if (typeof window === 'undefined') return fallback;
  const data = localStorage.getItem(key);
  if (!data) return fallback;
  try { return JSON.parse(data); } catch { return fallback; }
};

const saveToStorage = (key, data) => {
  if (typeof window !== 'undefined') {
    localStorage.setItem(key, JSON.stringify(data));
  }
};

          function toSolidBytes(match, p1) {
              return String.fromCharCode('0x' + p1);
      }));
      
      const shareUrl = `${window.location.origin}${window.location.pathname}?data=${encodedData}`;

        showToast("Tautan snapshot lengkap disalin!");
      } else {
        showToast("Gagal menyalin tautan.", "error");

      console.error("Share error:", error);
      showToast("Data terlalu besar/kompleks untuk tautan.", "error");

    setCategories([newCat, ...categories]);
    markAsDraft();
    showToast("Kategori ditambahkan");

    setCategories(prev => prev.map(c => c.id === id ? { ...c, [field]: value } : c));
    markAsDraft();
  };

    const newCats = [...categories];
    if (direction === 'up' && index > 0) {
      [newCats[index], newCats[index - 1]] = [newCats[index - 1], newCats[index]];
    } else if (direction === 'down' && index < newCats.length - 1) {
      [newCats[index], newCats[index + 1]] = [newCats[index + 1], newCats[index]];
    }
    setCategories(newCats);
    markAsDraft();

      }

        setConfirmAction(null);

    // Browser modern memblokir akses langsung ke file lokal demi keamanan.
    if (app.url.match(/^[a-zA-Z]:\\/) || app.url.startsWith('file://')) {
       // Tampilkan pesan edukasi yang jelas
       alert(
         "â›” AKSES DITOLAK BROWSER\n\n" +
         "Browser tidak mengizinkan akses langsung ke file lokal (C:\\...).\n" +
         "Solusi: Anda harus membuat 'Custom URI Scheme' di Windows Registry agar bisa dipanggil via protokol (misal: eziview://)."
       );
       return;

    // --- LOGIKA BARU: MEMBUKA APLIKASI PC (CUSTOM PROTOCOL) ---
    // Jika tidak dimulai dengan http/https, dianggap sebagai aplikasi lokal
    // Contoh: "calculator:", "ms-settings:", "zoommtg://", "edmi://"
    if (!app.url.startsWith('http://') && !app.url.startsWith('https://')) {
       try {
           // PERBAIKAN: Gunakan window.open('_blank') agar stabil memicu protokol
           // DAN HAPUS LOGIKA AUTO-CLOSE agar dialog aplikasi sempat muncul.
           const newWindow = window.open(app.url, '_blank');
           
           if (!newWindow || newWindow.closed || typeof newWindow.closed == 'undefined') {
               showToast("Popup diblokir! Izinkan pop-up agar aplikasi bisa terbuka.", "warning");
           } else {
               showToast(`Meluncurkan aplikasi lokal: ${app.name}`, "info");
               // Kita biarkan tab terbuka agar proses hand-off ke OS berhasil
           }
       } catch (e) {
           showToast("Gagal memanggil protokol. Pastikan sudah terdaftar di PC.", "error");

       return;

    const isGoogle = app.url.includes('drive.google.com') ||

    if (isGoogle) {

        return; // SELESAI, Masuk Tab
    }

    // 2. CEK WEBSITE LAIN (Buka di Browser Langsung / Tab Baru)
    // Untuk semua website non-Google (publik atau lokal)
    try {
        window.open(app.url, '_blank');
        showToast(`Membuka: ${app.name}`, "info");
    } catch (e) {
        showToast("Gagal membuka tautan. Pastikan pop-up diizinkan.", "error");

  };

    if (activeTabId === tabId) {

        setActiveTabId(newTabs[newTabs.length - 1].id);
      } else {
        setActiveTabId('dashboard');
      }

  const updateTabZoom = (tabId, delta, absoluteValue = null) => {
    setOpenTabs(prevTabs => prevTabs.map(tab => {
      if (tab.id === tabId) {
        let newZoom = absoluteValue !== null ? absoluteValue : tab.zoom + delta;
        newZoom = Math.min(Math.max(newZoom, 0.5), 3.0);
        return { ...tab, zoom: newZoom };
      }
      return tab;
    }));
  };

  const handleTouchStart = (e) => {
    if (e.touches.length === 2) {
      const dist = Math.hypot(
        e.touches[0].pageX - e.touches[1].pageX,
        e.touches[0].pageY - e.touches[1].pageY
      );
      touchStartDist.current = dist;
    }
  };

  const handleTouchMove = (e, tabId, currentZoom) => {
    if (e.touches.length === 2) {
      const dist = Math.hypot(
        e.touches[0].pageX - e.touches[1].pageX,
        e.touches[0].pageY - e.touches[1].pageY
      );
      
      if (touchStartDist.current > 0) {
        const delta = dist - touchStartDist.current;
        if (Math.abs(delta) > 20) {
           const zoomChange = delta > 0 ? 0.05 : -0.05;
           updateTabZoom(tabId, zoomChange);
           touchStartDist.current = dist; 
        }
      }
    }

         case 'za':

           return b.id - a.id;
         case 'oldest': 
           return a.id - b.id;

         default:

             return b.name.localeCompare(a.name, undefined, { numeric: true, sensitivity: 'base' });
           }
           
           if (!isNumA && !isNumB) {
             // Keduanya Teks: Urutkan A-Z
             return a.name.localeCompare(b.name, undefined, { numeric: true, sensitivity: 'base' });
           }

           // Campuran: Teks (isNum=false) duluan (-1), Angka (isNum=true) belakangan (1)
           return isNumA ? 1 : -1;
       }
    });

  const currentSortLabel = sortSettings[selectedCat] || 'hybrid';

  return (
    <div className="flex h-screen bg-[#FDFDFD] text-slate-900 font-sans overflow-hidden">

              if (json.config) {
                 if (json.config.githubUrl) setGithubUrl(json.config.githubUrl);

                 if (json.config.appTitle) setAppTitle(json.config.appTitle);
                 if (json.config.appLogo) setAppLogo(json.config.appLogo);
                 if (json.config.dashboardTitle) setDashboardTitle(json.config.dashboardTitle);

                 if (json.config.sortSettings) setSortSettings(json.config.sortSettings);

              if (json.version) setActiveVersion(json.version + "*");

            }

        };
        reader.readAsText(file);
      }} accept=".json" className="hidden" />

      {/* MOBILE OVERLAY BACKDROP */}
      {sidebarOpen && (
        <div 
          onClick={() => setSidebarOpen(false)}

      )}

        fixed inset-y-4 left-4 rounded-[2.5rem] bg-white border border-slate-100 flex flex-col transition-all duration-700 ease-[cubic-bezier(0.23,1,0.32,1)] z-50 shadow-[0_10px_50px_rgba(0,0,0,0.03)]
        ${sidebarOpen ? 'w-[280px] translate-x-0' : 'w-[90px] -translate-x-[200%] md:translate-x-0'}
      `}>
        <div className="h-28 flex items-center justify-center shrink-0 relative px-2">
          <div className="flex items-center gap-3">
            <div className="relative shrink-0 w-10 h-10 bg-indigo-600 rounded-xl flex items-center justify-center text-white shadow-xl shadow-indigo-100 transform rotate-3">
               {React.createElement(IconMap[appLogo] || Sparkles, { size: 20, fill: "currentColor" })}
            </div>
            {sidebarOpen && (
              <div className="flex flex-col justify-center">
                <span className="text-base font-extrabold tracking-tight leading-4 text-slate-800 max-w-[160px] text-left break-words">
                   {appTitle.includes('.') ? (
                      <>{appTitle.substring(0, appTitle.lastIndexOf('.'))}<span className="text-indigo-600">{appTitle.substring(appTitle.lastIndexOf('.'))}</span></>
                   ) : (
                      appTitle
                   )}
                </span>
                <span className="text-[8px] font-bold text-slate-400 uppercase tracking-[0.15em] mt-1 text-left">Enterprise System</span>
              </div>
            )}
          </div>
          <button 
            onClick={() => setSidebarOpen(false)}
            className={`md:hidden absolute right-5 top-1/2 -translate-y-1/2 p-2 bg-slate-50 text-slate-400 hover:text-rose-500 hover:bg-rose-50 rounded-full transition-all ${!sidebarOpen && 'hidden'}`}
          >
            <X size={18} />
          </button>
        </div>
        
        <div className="flex-1 px-5 py-2 space-y-2 overflow-y-auto custom-scrollbar">
          <button onClick={() => { setActiveTabId('dashboard'); setSelectedCat('all'); }} className={`w-full flex items-center p-4 rounded-2xl transition-all duration-500 group ${selectedCat === 'all' ? 'bg-indigo-50 text-indigo-600 shadow-sm' : 'text-slate-400 hover:bg-slate-50 hover:text-slate-600'}`}>
            <LayoutGrid size={22} />
            {sidebarOpen && <span className="ml-4 text-sm font-bold">Semua Aplikasi</span>}

          <div className="pt-8 pb-4">
            <p className={`px-5 mb-4 text-[9px] font-black text-slate-200 uppercase tracking-[0.3em] ${!sidebarOpen && 'hidden'}`}>Kategori Layanan</p>

'bg-indigo-50 text-indigo-600 shadow-sm' : 'text-slate-400 hover:bg-slate-50 hover:text-slate-600'}`}>

           <div className={`rounded-3xl bg-slate-50 p-1.5 flex flex-col gap-1 ${!sidebarOpen && 'items-center'}`}>
              <button onClick={() => setShowHistory(!showHistory)} className="p-3.5 rounded-2xl hover:bg-white hover:shadow-sm text-slate-400 hover:text-indigo-600 transition-all flex items-center gap-3">
                <History size={18} />

                <Settings size={18} />
                {sidebarOpen && <span className="text-xs font-bold">Konfigurasi</span>}

           </div>

        {/* Header Glassmorphism */}
        <header className="px-8 pt-6 pb-2 sticky top-0 z-40">
           <div className="bg-white/80 backdrop-blur-2xl border border-white/40 rounded-[2.5rem] shadow-[0_15px_40px_rgba(0,0,0,0.03)] px-8 py-4 flex items-center justify-between gap-8">

'bg-orange-50 border-orange-100 text-orange-600 hover:shadow-xl hover:shadow-orange-50' : 'bg-emerald-50 border-emerald-50 text-emerald-600'}`}
                 >

                    <span>v{activeVersion}</span>
                    {activeVersion.includes('*') && <Save size={16} />}

           <div className="flex items-center gap-2 overflow-x-auto no-scrollbar pb-1">
              {/* Home / Dashboard Tab */}
              <button 
                 onClick={() => setActiveTabId('dashboard')}

                          value={githubUrl} 
                          onChange={(e) => setGithubUrl(e.target.value)} 
                          className="w-full px-6 py-4 rounded-2xl border-none bg-white shadow-sm text-sm font-bold focus:ring-4 focus:ring-indigo-50 outline-none transition-all" 
                          placeholder="https://github.com/..." 
                       />
                    </div>
                    <div className="space-y-3">
                       <label className="text-[10px] font-black text-slate-400 ml-3 uppercase tracking-widest">Personal Access Token</label>
                       <input type="password" value={githubToken} onChange={(e) => setGithubToken(e.target.value)} className="w-full px-6 py-4 rounded-2xl border-none bg-white shadow-sm text-sm font-bold focus:ring-4 focus:ring-indigo-50 outline-none transition-all" placeholder="ghp_****************" />
                    </div>
                    
                    {/* NEW: Upload Authorization Code (Removed from Settings UI, kept in logic) */}
                    
                    <div className="flex gap-4 pt-4">
                       <button onClick={() => fetchFromGithub(githubUrl)} className="flex-1 py-4 bg-white text-slate-700 font-black text-xs rounded-2xl shadow-sm hover:shadow-xl hover:-translate-y-1 transition-all border border-slate-100 flex items-center justify-center gap-3"><RefreshCw size={16} /> TARIK DATA</button>
                       <button onClick={initiateUpload} className="flex-1 py-4 bg-indigo-600 text-white font-black text-xs rounded-2xl shadow-xl shadow-indigo-100 hover:shadow-indigo-200 hover:-translate-y-1 transition-all flex items-center justify-center gap-3"><Send size={16} /> UNGGAH DATA</button>
                    </div>
                 </div>
              </section>
            </div>
          </div>
        </div>
      )}
      
      {/* AUTH POPUP MODAL (NEW) */}
      {showAuthModal && (
        <div className="fixed inset-0 bg-slate-900/60 backdrop-blur-sm z-[200] flex items-center justify-center p-4 animate-in fade-in duration-200">
          <div className="bg-white rounded-[2rem] shadow-2xl w-full max-w-sm p-8 animate-in zoom-in-95 duration-300 border border-slate-100 text-center">
            <div className="w-16 h-16 bg-indigo-100 text-indigo-600 rounded-3xl flex items-center justify-center mb-6 mx-auto shadow-lg shadow-indigo-50">
               <Lock size={32} />
            </div>
            <h3 className="text-xl font-black text-slate-800 mb-2">Otorisasi Diperlukan</h3>
            <p className="text-xs text-slate-500 font-medium leading-relaxed mb-6">Masukkan kode akses untuk melanjutkan proses unggah data ke cloud.</p>
            
            <input 
               type="password" 
               value={authInput}
               onChange={(e) => setAuthInput(e.target.value)}
               className="w-full bg-slate-50 border-none rounded-2xl px-6 py-4 text-center text-lg font-bold tracking-widest mb-6 focus:ring-4 focus:ring-indigo-100 transition-all outline-none"
               placeholder="â€¢ â€¢ â€¢"
               autoFocus
            />

            <div className="flex gap-3">
               <button onClick={() => { setShowAuthModal(false); setAuthInput(''); }} className="flex-1 py-3 bg-slate-100 text-slate-600 font-bold rounded-xl hover:bg-slate-200 transition-colors text-xs">Batal</button>
               <button onClick={handleAuthSubmit} className="flex-1 py-3 bg-indigo-600 text-white font-bold rounded-xl hover:bg-indigo-700 transition-colors shadow-lg shadow-indigo-200 text-xs flex items-center justify-center gap-2">
                  <Unlock size={14} /> Buka Akses
               </button>
            </div>
          </div>
        </div>
      )}

      {/* Editing Drawer / Modal */}
      {editing && (
        <div className="fixed inset-0 bg-slate-900/50 backdrop-blur-sm z-[110] flex items-center justify-center p-6">
          <div className="bg-white rounded-[3rem] shadow-4xl w-full max-w-xl p-10 animate-in zoom-in-95 duration-300 max-h-[90vh] overflow-y-auto custom-scrollbar">
            <div className="flex items-center gap-4 mb-10">
               <div className="w-1.5 h-10 bg-indigo-600 rounded-full"></div>
               <h2 className="text-3xl font-black text-slate-800 tracking-tighter">Edit Layanan</h2>
            </div>
            
            <div className="space-y-6">
              
              {/* --- BAGIAN BARU: SELECTOR IKON & WARNA --- */}
              <div className="space-y-3">
                <label className="text-[10px] font-black text-slate-300 ml-2 uppercase tracking-widest">Tampilan Visual</label>
                <div className="flex gap-5 items-start">
                  
                  {/* Icon Selector (VISUAL PICKER) */}
                  <div className="relative shrink-0 z-50">
                     <button 
                        onClick={() => setShowIconPicker(!showIconPicker)}
                        className={`w-24 h-24 rounded-3xl bg-gradient-to-br ${editing.color || 'from-slate-800 to-slate-950'} flex items-center justify-center text-white shadow-xl shadow-slate-200 transition-all hover:scale-[1.02] relative group border-4 border-white`}
                        title="Klik untuk ganti ikon"
                     >
                        {React.createElement(IconMap[editing.icon] || LayoutGrid, { size: 40, strokeWidth: 1.5 })}
                        <div className="absolute -bottom-2 -right-2 bg-white text-slate-800 p-2 rounded-full shadow-lg border border-slate-50 z-10 group-hover:scale-110 transition-transform">
                           <RefreshCw size={14} />
                        </div>
                     </button>

                     {/* Popover Grid Icons */}
                     {showIconPicker && (
                        <>
                          <div className="fixed inset-0 z-40" onClick={() => setShowIconPicker(false)}></div>
                          <div className="absolute top-full left-0 mt-3 p-3 bg-white rounded-2xl shadow-[0_10px_40px_-10px_rgba(0,0,0,0.2)] border border-slate-100 z-50 w-[300px] grid grid-cols-5 gap-2 animate-in fade-in zoom-in-95 duration-200 h-64 overflow-y-auto custom-scrollbar">
                              {Object.keys(IconMap).map(key => (
                                <button
                                    key={key}
                                    onClick={() => { setEditing({...editing, icon: key}); setShowIconPicker(false); }}
                                    className={`aspect-square rounded-xl flex items-center justify-center transition-all ${editing.icon === key ? 'bg-indigo-600 text-white shadow-md shadow-indigo-200 scale-105' : 'bg-slate-50 text-slate-400 hover:bg-indigo-50 hover:text-indigo-600 hover:scale-110'}`}
                                    title={key}
                                >
                                    {React.createElement(IconMap[key], { size: 20 })}
                                </button>
                              ))}
                          </div>
                        </>
                     )}
                  </div>

                  {/* Color Picker Grid (COMPACT VERSION) */}
                  <div className="flex-1">
                     <p className="text-xs text-slate-400 font-bold mb-3 ml-1">Pilih Tema Warna</p>
                     <div className="grid grid-cols-9 gap-1.5">
                       {[
                         // Classic
                         'from-blue-500 to-indigo-600', 
                         'from-emerald-500 to-teal-600', 
                         'from-orange-500 to-red-600', 
                         'from-rose-500 to-pink-600',
                         'from-violet-500 to-purple-600', 
                         'from-slate-700 to-slate-900',
                         // New Vibrant Options
                         'from-cyan-500 to-blue-600',
                         'from-amber-400 to-orange-500',
                         'from-lime-500 to-green-600',
                         'from-fuchsia-600 to-pink-600',
                         'from-sky-500 to-indigo-500',
                         'from-indigo-500 to-purple-500',
                         // Dark & Elegant
                         'from-gray-800 to-black',
                         'from-slate-600 to-zinc-800',
                         'from-neutral-700 to-stone-800',
                         // Bright & Fresh
                         'from-teal-400 to-emerald-600',
                         'from-red-500 to-rose-700',
                         'from-pink-500 to-rose-500'
                       ].map((c, i) => (
                         <button 
                           key={i} 
                           onClick={() => setEditing({...editing, color: c})}
                           // PERUBAHAN: aspect-square diubah menjadi h-6 (tinggi fix 24px) agar lebih kecil dan pas
                           className={`w-full h-6 rounded-md bg-gradient-to-br ${c} ${editing.color === c ? 'ring-2 ring-offset-1 ring-slate-900 scale-90' : 'hover:scale-110 hover:shadow-md'} transition-all`}
                           title="Pilih warna"
                         />
                       ))}
                     </div>
                  </div>
                </div>
              </div>
              {/* --- END BAGIAN BARU --- */}

              <div className="space-y-3">
                <label className="text-[10px] font-black text-slate-300 ml-2 uppercase tracking-widest">Label Aplikasi</label>
                <input type="text" value={editing.name} onChange={e => setEditing({...editing, name: e.target.value})} className="w-full bg-slate-50 border-none rounded-2xl px-6 py-4 text-sm font-black focus:bg-white focus:ring-4 focus:ring-indigo-50 outline-none transition-all" />
              </div>
              <div className="space-y-3">
                <label className="text-[10px] font-black text-slate-300 ml-2 uppercase tracking-widest">Endpoint URL / Protokol Aplikasi</label>
                <input 
                  type="text" 
                  value={editing.url} 
                  onChange={e => setEditing({...editing, url: e.target.value})} 
                  className="w-full bg-slate-50 border-none rounded-2xl px-6 py-4 text-sm font-bold focus:bg-white focus:ring-4 focus:ring-indigo-50 outline-none transition-all" 
                  placeholder="https://... atau calculator: atau edmi://"
                />
                <p className="text-[9px] text-slate-400 ml-2 italic">
                   Google Drive/Docs: Buka di Tab Aplikasi<br/>
                   Website Lain: Buka di Browser (Tab Baru)<br/>
                   App PC: <b>eziview://</b> (Buka Langsung)
                </p>
              </div>
              <div className="space-y-3">
                 <label className="text-[10px] font-black text-slate-300 ml-2 uppercase tracking-widest">Kategori</label>
                 <div className="flex gap-3">
                    {categories.map(c => (
                       <button key={c.id} onClick={() => setEditing({...editing, category: c.id})} className={`flex-1 py-3.5 rounded-2xl text-[10px] font-black border-2 transition-all ${editing.category === c.id ? 'bg-indigo-600 text-white border-indigo-600 shadow-lg shadow-indigo-100' : 'bg-white text-slate-400 border-slate-100 hover:border-slate-200'}`}>
                          {c.label.toUpperCase()}
                       </button>
                    ))}
                 </div>
              </div>
              <div className="space-y-3">
                <label className="text-[10px] font-black text-slate-300 ml-2 uppercase tracking-widest">Ringkasan Layanan</label>
                {/* PERUBAHAN: h-14 dikembalikan menjadi h-20 */}
                <textarea 
                  value={editing.desc} 
                  onChange={e => setEditing({...editing, desc: e.target.value})} 
                  className="w-full bg-slate-50 border-none rounded-2xl px-6 py-4 text-sm font-bold h-20 resize-none focus:bg-white focus:ring-4 focus:ring-indigo-50 outline-none transition-all" 
                />
              </div>
            </div>
            <div className="mt-12 flex gap-4">
              <button onClick={handleDeleteApp} className="px-8 py-4 text-rose-500 font-black text-[10px] uppercase tracking-widest hover:bg-rose-50 rounded-2xl transition-all">Hapus</button>
              <div className="flex-1"></div>
              <button onClick={() => setEditing(null)} className="px-8 py-4 text-slate-400 font-black text-[10px] uppercase tracking-widest hover:bg-slate-50 rounded-2xl transition-all">Batal</button>
              
              {/* FIX: Logika Simpan Pintar (Update jika ada, Tambah Baru jika belum ada) */}
              <button onClick={() => { 
                const exists = apps.some(a => a.id === editing.id);
                if (exists) {
                   // Update data lama
                   setApps(prev => prev.map(x => x.id === editing.id ? editing : x));
                } else {
                   // Tambah data baru
                   setApps(prev => [...prev, editing]);
                }
                markAsDraft(); 
                setEditing(null); 
              }} className="px-10 py-4 bg-slate-900 text-white rounded-2xl font-black text-[10px] uppercase tracking-widest shadow-2xl shadow-slate-200 hover:bg-indigo-600 hover:scale-105 transition-all">Konfirmasi</button>
            </div>
          </div>
        </div>
      )}

      {/* Global Confirmation Modal */}
      {confirmAction && (
        <ConfirmModal 
          isOpen={!!confirmAction} 
          title={confirmAction.title}
          message={confirmAction.message}
          onConfirm={confirmAction.onConfirm}
          onCancel={() => setConfirmAction(null)}
        />
      )}
      
      {toast && <Toast message={toast.msg} type={toast.type} />}
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Plus+Jakarta+Sans:wght@400;500;600;700;800&display=swap');
        body { font-family: 'Plus Jakarta Sans', sans-serif; letter-spacing: -0.01em; }
        .custom-scrollbar::-webkit-scrollbar { width: 4px; height: 4px; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #E2E8F0; border-radius: 10px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .no-scrollbar::-webkit-scrollbar { display: none; }
        .no-scrollbar { -ms-overflow-style: none; scrollbar-width: none; }
        
        /* Animation Keyframes for Marquee */
        @keyframes marquee {
          0% { transform: translateX(0); }
          100% { transform: translateX(-50%); }
        }
        .animate-marquee {
          animation: marquee 30s linear infinite;
        }
        /* Pause on hover for readability */
        .animate-marquee:hover {
          animation-play-state: paused;
        }
      `}</style>
    </div>
  );
}
